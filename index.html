1.git init //for initialization of empty repository 2.git add filename // for moving the file to staging area 3.git commit -m "message" // commit the file to the local repository 4.git commit -a -m "message" //direct stage+commit 5.git diff //for getting the changes made compared to staging area 6.gid diff --staged //for getting changes made compared to local repository 7.git rm --cached filename //for removing a file from repository 8.ssh-keygen -o //for generating ssh 9.git branch -M main //for creating a main branch 10.git remote add origin "ssh link" //for connecting to origin 11.git push -u origin main //for pushing the data to github
